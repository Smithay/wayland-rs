79pub fn snake_to_camel(input: &str) -> String {
80    let result = input
81        .split('_')
82        .flat_map(|s| {
83            let mut first = true;
84            s.chars().map(move |c| {
85                if first {
86                    first = false;
87                    c.to_ascii_uppercase()
88                } else {
89                    c
90                }
91            })
92        })
93        .collect::<String>();
94
95    if is_camel_keyword(&result) {
96        format!("_{}", &result)
97    } else {
98        result
99    }
100}

102pub fn dotted_to_relname(input: &str) -> TokenStream {
103    let mut it = input.split('.');
104    match (it.next(), it.next()) {
105        (Some(module), Some(name)) => {
106            let module = Ident::new(module, Span::call_site());
107            let ident = Ident::new(&snake_to_camel(name), Span::call_site());
108            quote::quote!(super::#module::#ident)
109        }
110        (Some(name), None) => {
111            Ident::new(&snake_to_camel(name), Span::call_site()).into_token_stream()
112        }
113        _ => unreachable!(),
114    }
